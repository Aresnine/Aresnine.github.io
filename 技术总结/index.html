```javascript
const UserContext = createContext()

  setup() {
    return {
      user: {
        name: 'Tom',
        age: 18
      }
    }
  },

      user: this.user
    }
  },

    Child
  }

  setup() {

    return {
      user
    }
  }

  setup() {
    return {
      UserContext
    }
  },
  components: {
    Parent
  }
})
```
